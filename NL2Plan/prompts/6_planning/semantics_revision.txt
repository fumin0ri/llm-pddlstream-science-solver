(define (domain wildlife)
    (:requirements

        :universal-preconditions :conditional-effects
    )

    (:types
        ranger - object ; A worker protecting the animals.
        location - object ; Places which can be traveled between.
        habitat - location ; Habitats are a type of location.
        animal - object ; The animals living in the park.
    )

    (:predicates
        (at ?o - object ?l - location) ; True if the object ?o (an animal or a ranger) is at the location ?l. Properties: MAXONCE(?o)
        (guarded ?l - location) ; True if the location ?l is guarded by anyone.
        (guarding ?r - ranger) ; True if the ranger ?r is guarding anything.
        (neighboring ?l1 - location ?l2 - location) ; True if the location ?l1 is neighboring the location ?l2. Properties: SYMMETRICAL(?l1, ?l2)
    )

    (:action walk
        :parameters (
            ?r - ranger
            ?l1 - location
            ?l2 - location
        )

        :precondition (and
            (at ?r ?l1)
            (neighboring ?l1 ?l2)
            (not (guarding ?r))
        )

        :effect (and
            (not (at ?r ?l1))
            (at ?r ?l2)
        )
    )

    (:action move_animal
        :parameters (
            ?r - ranger
            ?a - animal
            ?h1 - habitat
            ?h2 - habitat
        )

        :precondition (and
            (at ?r ?h1)
            (at ?a ?h1)
            (neighboring ?h1 ?h2)
            (not (guarding ?r))
        )

        :effect (and
            (not (at ?r ?h1))
            (not (at ?a ?h1))
            (at ?a ?h2)
            (at ?r ?h2)
        )
    )

    (:action guard
        :parameters (
            ?r - ranger
            ?l - location
        )

        :precondition (and
            (at ?r ?l)
            (not (guarded ?l))
        )

        :effect (and
            (guarding ?r)
            (guarded ?l)
        )
    )

    (:action unguard
        :parameters (
            ?r - ranger
            ?l - location
        )

        :precondition (and
            (at ?r ?l)
            (guarding ?r)
        )

        :effect (and
            (not (guarding ?r))
            (not (guarded ?l))
        )
    )
)
```
