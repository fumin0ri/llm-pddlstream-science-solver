    (:requirements
        :strips :typing :equality :negative-preconditions :disjunctive-preconditions
        :universal-preconditions :conditional-effects
    )

        floor - object ; The floor where elevators and people can be
        person - object ; A person traveling in the building
    )

    (:predicates

    (:action enter
        :parameters (
            ?p - person
            ?e - elevator

            (at ?e ?f)
        )

        :effect (and
            (not (at ?p ?f))
            (in ?p ?e)
        )
    )

    (:action exit

            ?p - person
            ?e - elevator
            ?f - floor

            (at ?e ?f)

        :effect (and
